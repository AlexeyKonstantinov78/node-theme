pwd                             - посмотреть где в каком месте находимся
mkdir                           - создает какатлог
mkdir [каталог/подкаталог] -p   - для создания подкаталога
ls                              - список файлов
ls -l                           - список файлов раширенный
ls -al                          - список файлов раширенный показывает скрытие файлы
cd                              - переход в каталог
cd ..                           -  возврат на каталог выше
touch                           - создание файла
> имя файла                     - сокращенное команда создание файла
cat                             - просмотр содержимого файла
rm                              - удаление файла
rm [имя катоалоога] -rf         - удаление катоалоога
node                            - запусл REPL среда движка 
code .                          - запуск редактора в текушем каталоге
wim                             - запуск редактора wim
nano                            - запуск редактора nano
